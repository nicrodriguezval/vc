linkGrayscale

The grayscale is a kind of image which represents the amount of light of every pixel of the whole image. The image is composed exclusively of shades of gray. The gray color is one in which the RGB channels have got the same or a similar quantity. To transform an image to grayscale, we calculated the average of the components Red, Green and Blue of each pixel, but we can also do this using this equation: Red0.3+Green0.59+Blue*0.11.

shader.vert
1link// Precision seems mandatory in webgl

2linkprecision highp float;

3link

4link// 1. Attributes and uniforms sent by p5.js

5link

6link// Vertex attributes and some uniforms are sent by

7link// p5.js following these naming conventions:

8link// https://github.com/processing/p5.js/blob/main/contributor_docs/webgl_mode_architecture.md

9link

10link// 1.1. Attributes

11link// vertex position attribute

12linkattribute vec3 aPosition;

13link

14link// vertex texture coordinate attribute

15linkattribute vec2 aTexCoord;

16link

17link// vertex color attribute

18linkattribute vec4 aVertexColor;

19link

20link// 1.2. Matrix uniforms

21link

22link// The vertex shader should project the vertex position into clip space:

23link// vertex_clipspace = vertex * projection * view * model (see the gl_Position below)

24link// Details here: http://visualcomputing.github.io/Transformations

25link

26link// Either a perspective or an orthographic projection

27linkuniform mat4 uProjectionMatrix;

28link

29link// modelview = view * model

30linkuniform mat4 uModelViewMatrix;

31link

32link// B. varying variable names are defined by the shader programmer:

33link// vertex color

34linkvarying vec4 vVertexColor;

35link

36link// vertex texcoord

37linkvarying vec2 vTexCoord;

38link

39linkvoid main() {

40link// copy / interpolate color

41linkvVertexColor = aVertexColor;

42link// copy / interpolate texcoords

43linkvTexCoord = aTexCoord;

44link// vertex projection into clipspace

45linkgl_Position = uProjectionMatrix * uModelViewMatrix * vec4(aPosition, 1.0);

46link}

grayscale.frag
1linkprecision mediump float;

2link

3link// texture is sent by the sketch

4linkuniform sampler2D texture;

5link

6linkuniform float RGBval[3];

7linkuniform int mode;

8link

9link// interpolated color (same name and type as in vertex shader)

10linkvarying vec4 vVertexColor;

11link// interpolated texcoord (same name and type as in vertex shader)

12linkvarying vec2 vTexCoord;

13link

14linkvoid main(){

15link // texture2D(texture, vTexCoord) samples texture at vTexCoord

16link // and returns the normalized texel color

17link // texel color times vVertexColor gives the final normalized pixel color

18link

19link vec4 color = texture2D(texture, vTexCoord) * vVertexColor;

20link //float gray = dot(color.rgb /*color.xyz*/, vec3(1.0 / 3.0, 1.0 / 3.0, 1.0 / 3.0));

21link float gray = dot(color.rgb /*color.xyz*/, vec3(RGBval[0],RGBval[1],RGBval[2]));

22link vec4 negColor = vec4(vec3(1.0) - vec3(color.r,color.g,color.b),1.0);

23link if(mode == 0){ //Original color mode

24link   gl_FragColor = color;

25link } else if(mode == 1){ //Grayscale mode

26link   gl_FragColor = vec4(vec3(gray), 1.0);

27link } else if(mode == 2){ //Negative mode

28link   gl_FragColor = negColor;

29link }

30link}
