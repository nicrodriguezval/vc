150linkfunction updateFPS(){

151link let decimalMillis = millis() % 1000;

152link let fps = 0;

153link if(decimalMillis < savedMillis){

154link   savedMillis = decimalMillis;

155link   fps = savedFPS + frameCount;

156link   fpsButton.html(fps);

157link   savedFPS = -frameCount;

158link } else {

159link   savedMillis = decimalMillis;

160link }

217linkfunction cover(texture = false){

218link beginShape();

219link if(texture){

220link   vertex(-width / 2, -height /2, 0, 0, 0);

221link   vertex( width / 2 - rightOffset, -height /2, 0, 1, 0);

222link   vertex( width / 2 - rightOffset, height /2, 0, 1, 1);

223link   vertex( -width / 2, height /2, 0, 0, 1);

224link } else {

225link   vertex(-width / 2, -height /2, 0);

226link   vertex( width / 2 - rightOffset, -height /2, 0);

227link   vertex( width / 2 - rightOffset, height /2, 0);

228link   vertex( -width / 2, height /2, 0);

229link }

230link endShape(CLOSE);

231link}
