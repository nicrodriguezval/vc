linkKernel (image processing)

To make image processing we need to use a small matrix that uses different values in it. This matrix is called Kernel. There are lots of effects such as blur, bottom sobel, emboss, identity, etc.

shader.vert
1link// Precision seems mandatory in webgl

2linkprecision highp float;

3link

4link// 1. Attributes and uniforms sent by p5.js

5link

6link// Vertex attributes and some uniforms are sent by

7link// p5.js following these naming conventions:

8link// https://github.com/processing/p5.js/blob/main/contributor_docs/webgl_mode_architecture.md

9link

10link// 1.1. Attributes

11link// vertex position attribute

12linkattribute vec3 aPosition;

13link

14link// vertex texture coordinate attribute

15linkattribute vec2 aTexCoord;

16link

17link// vertex color attribute

18linkattribute vec4 aVertexColor;

19link

20link// 1.2. Matrix uniforms

21link

22link// The vertex shader should project the vertex position into clip space:

23link// vertex_clipspace = vertex * projection * view * model (see the gl_Position below)

24link// Details here: http://visualcomputing.github.io/Transformations

25link

26link// Either a perspective or an orthographic projection

27linkuniform mat4 uProjectionMatrix;

28link

29link// modelview = view * model

30linkuniform mat4 uModelViewMatrix;

31link

32link// B. varying variable names are defined by the shader programmer:

33link// vertex color

34linkvarying vec4 vVertexColor;

35link

36link// vertex texcoord

37linkvarying vec2 vTexCoord;

38link

39linkvoid main() {

40link// copy / interpolate color

41linkvVertexColor = aVertexColor;

42link// copy / interpolate texcoords

43linkvTexCoord = aTexCoord;

44link// vertex projection into clipspace

45linkgl_Position = uProjectionMatrix * uModelViewMatrix * vec4(aPosition, 1.0);

46link}

28link

29link   // set the output color, it’s RGBA 0-1, not 0-255.

32link   vec2 st = gl_FragCoord.xy/resolution;

33link   float y = mouse.y * st.y;

34link   float x = st.x * mouse.x;

35link   // vec3 color = vec3(x,y,0.0);

47link   vec2 tc8 = vTexCoord + vec2(+steps.x, +steps.y);

52link   vec4 col3 = texture2D(texture, tc3);

53link   vec4 col4 = texture2D(texture, tc4);

58link

59link   vec4 sumku = vec4(ku.x*col6) + vec4(ku.y*col7) + vec4(ku.z*col8);

60link   vec4 sumkm = vec4(km.x*col3) + vec4(km.y*col4) + vec4(km.z*col5);

61link   vec4 sumkd = vec4(kd.x*col0) + vec4(kd.y*col1) + vec4(kd.z*col2);

62link

63link

64link   vec4 sum = vec4(sumku) + vec4(sumkm) + vec4(sumkd);

65link   sum.w = 1.0;

66link

67link   gl_FragColor = vec4(sum) * vVertexColor;

68link   // gl_FragColor = texture2D(texture, vTexCoord) * vVertexColor;

70link}

texture.frag
1linkprecision mediump float;

2link

3link// texture is sent by the sketch

4linkuniform sampler2D texture;

5link

6linkuniform float textureWidth;

7linkuniform float textureHeight;

8linkuniform float kernel[9];

9link// interpolated color (same name and type as in vertex shader)

10linkvarying vec4 vVertexColor;

11link// interpolated texcoord (same name and type as in vertex shader)

12linkvarying vec2 vTexCoord;

13link

14linkvoid main(){

15link// texture2D(texture, vTexCoord) samples texture at vTexCoord

16link// and returns the normalized texel color

17link// texel color times vVertexColor gives the final normalized pixel color

18link

19link// vec4 sum = vec4(0.0);

20linkfloat stepSizeX=1.0/textureWidth;

21linkfloat stepSizeY=1.0/textureHeight;

22link

23link//vec4 color = texture2D(texture, vTexCoord) * vVertexColor;

24link//float gray = dot(color.rgb /*color.xyz*/, vec3(0.333, 0.333, 0.333));

25link//gl_FragColor = vec4(vec3(gray), 1.0);

26link

27linkvec2 tc0=vTexCoord+vec2(-stepSizeX,-stepSizeY);

28linkvec4 col0=texture2D(texture,tc0)*kernel[0];

29linkvec2 tc1=vTexCoord+vec2(0.0,-stepSizeY);

30linkvec4 col1=texture2D(texture,tc1)*kernel[1];

31linkvec2 tc2=vTexCoord+vec2(stepSizeX,-stepSizeY);

32linkvec4 col2=texture2D(texture,tc2)*kernel[2];

33linkvec2 tc3=vTexCoord+vec2(-stepSizeX,0.);

34linkvec4 col3=texture2D(texture,tc3)*kernel[3];

35linkvec2 tc4=vTexCoord+vec2(0.0,0.0);

36linkvec4 col4=texture2D(texture,tc4)*kernel[4];

37linkvec2 tc5=vTexCoord+vec2(stepSizeX,0.0);

38linkvec4 col5=texture2D(texture,tc5)*kernel[5];

39linkvec2 tc6=vTexCoord+vec2(-stepSizeX,stepSizeY);

40linkvec4 col6=texture2D(texture,tc6)*kernel[6];

41linkvec2 tc7=vTexCoord+vec2(0.0,stepSizeY);

42linkvec4 col7=texture2D(texture,tc7)*kernel[7];

43linkvec2 tc8=vTexCoord+vec2(stepSizeX,stepSizeY);

44linkvec4 col8=texture2D(texture,tc8)*kernel[8];

45link

46linkvec4 sum=col0+col1+col2+col3+col4+col5+col6+col7+col8;

47link

48linkgl_FragColor=vec4(vec3(sum),1.)*vVertexColor;

49link}
