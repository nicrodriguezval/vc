linkProblem statement

This is an adaptation of the Mosaic sketch shown in the software implementations of Image Processing. The purpose of this sketch is to compare it to the hardware sketch, in order to have metrics about its performance.

60linkfunction processMosaic(){

61link

62link for (let i = 0; i < allImages.length; i++) {

63link   avg = 0;

64link

65link   for (let j = 0; j < allImages[i].width; j++) {

66link     for (let k = 0; k < allImages[i].height; k++) {

67link       avg += brightness(allImages[i].get(j, k));

68link     }

69link   }

70link

71link   bright[i] = avg / (allImages[i].width * allImages[i].height);

72link }

73link

74link for (let i = 0; i < brightImages.length; i++) {

75link   minDiff = 256;

76link

77link   for (let j = 0; j < bright.length; j++) {

78link     diff = abs(i - bright[j]);

79link

80link     if (diff < minDiff) {

81link       minDiff = diff;

82link       brightImages[i] = allImages[j];

83link     }

84link   }

85link }

86link

87link smaller = createImage(img.width / scl, img.height);

88link smaller.copy(img, 0, 0, img.width, img.height, 0, 0, img.width / scl, img.height / scl);

89link}

93link   for (let j = 0; j < (img.width / scl); j++) {

94link     index = int(brightness(smaller.get(i, j)));

95link     image(brightImages[index], i * scl, j * scl, scl, scl);

96link   }

97link }

98link}
