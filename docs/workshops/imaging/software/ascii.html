linkASCII art

ASCII art is a technique that represents an image into characters of the ASCII standard. This technique of representing images using characters is old, one of the first representations we know is "Simmias de Rodas. El hacha, ca. 300 a. C." which looks like the below image.

We represented Lenna's photo using this technique.

15linkfunction monospaceChar(content, size, posX, posY){

16link   fill(0);

17link   textSize(size);

18link   textFont("monospace");

19link   textStyle(BOLD)

20link   text(content,posX,posY);

21link}

23linkfunction yuvGrayscale(imag){

24link   let copied = createGraphics(imag.width,imag.height);

25link   copied.loadPixels();  

26link   for(let i = 0; i < imag.width; i++) {

27link     for(let j = 0; j < imag.height; j++) {

28link       //Y' component in YUV (below this line) is equivalent to LUMA

29link       copied.set(i, j, color(0.299*red(imag.get(i, j)) +  0.587*green(imag.get(i, j)) + 0.114*blue(imag.get(i, j))));

30link     }

31link   }

32link   copied.updatePixels();

33link   return copied;

34link }

35link

36link// Opacity information taken from https://observablehq.com/@grantcuster/sort-font-characters-by-percent-of-black-pixels

37linkfunction getSimilarChar26(number){

38link let ordered = "@$M%HKmd5Xah1ytnziLv+\":\'-." //26 ASCII elements, ordered by opacity

39link return ordered.substr(Math.floor(number/10),1) //Returns the corresponding string character for determined opacity of the pixel

40link}

41link

42linkfunction toASCII26(imag, size, initialX , initialY){

43link for(let i = 0; i < imag.width; i++) {

44link   for(let j = 0; j < imag.height; j++) {

45link     monospaceChar(getSimilarChar26(red(imag.get(i, j))),size,0.9*size*i + initialX,0.9*size*j + initialY); //Since the image is already gray, all rgb channels have the same value, we can pick any value

46link   }

47link }

48link}
